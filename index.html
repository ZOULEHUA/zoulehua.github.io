import tkinter as tk from tkinter import messagebox, ttk import random import time class Gobang: def __init__(self): self.root = tk.Tk() self.root.title("智能五子棋-难度可选版") # 游戏参数 self.board_size = 19 self.cell_size = 30 self.board_color = "#FFF8DC" self.mode = "human" self.game_active = False self.ai_difficulty = 5 self.history = [] self.board = [[0 for _ in range(19)] for _ in range(19)] # 比赛参数 self.player_scores = [0, 0] self.match_rules = {"三局两胜": 2, "五局三胜": 3, "十局六胜": 5} self.current_rule = "三局两胜" self.round_count = 0 # AI参数 self.ai_thinking = False self.difficulty_levels = { 1: {"search_depth": 0, "random_factor": 0.9}, 2: {"search_depth": 0, "random_factor": 0.7}, 3: {"search_depth": 1, "random_factor": 0.5}, 4: {"search_depth": 1, "random_factor": 0.3}, 5: {"search_depth": 2, "random_factor": 0.2}, 6: {"search_depth": 2, "random_factor": 0.1}, 7: {"search_depth": 3, "random_factor": 0.05}, 8: {"search_depth": 3, "random_factor": 0.02}, 9: {"search_depth": 4, "random_factor": 0.01}, 10: {"search_depth": 4, "random_factor": 0.0} } # 初始化界面 self.setup_gui() self.init_game() def setup_gui(self): # 控制面板 control_frame = tk.Frame(self.root) control_frame.pack(pady=10) # 模式切换 self.mode_btn = tk.Button(control_frame, text="切换AI模式", command=self.toggle_mode) self.mode_btn.pack(side=tk.LEFT, padx=5) # 难度选择 self.difficulty_frame = values=list(self.match_rules.keys()), state="readonly") self.rule_combobox.current(0) self.rule_combobox.pack(side=tk.LEFT, padx=5) self.rule_combobox.bind("<>", self.change_rule) # 分数显示 self.score_label = tk.Label(control_frame, text="分数 | 玩家1: 0 玩家2: 0", font=('微软雅黑', 12)) self.score_label.pack(side=tk.LEFT, padx=10) # 游戏控制 self.start_btn = tk.Button(control_frame, text="新游戏", command=self.start_new_game) self.start_btn.pack(side=tk.LEFT, padx=5) self.undo_btn = tk.Button(control_frame, text="悔棋", command=self.undo_move) self.undo_btn.pack(side=tk.LEFT, padx=5) # 棋盘 self.canvas = tk.Canvas( self.root, width=self.board_size * self.cell_size, height=self.board_size * self.cell_size, bg=self.board_color ) self.canvas.pack() self.draw_board() self.canvas.bind("", self.click_handler) def click_handler(self, event): if not self.game_active or self.ai_thinking: return grid_size = self.cell_size offset = grid_size // 2 x = int((event.x - offset) / grid_size) y = int((event.y - offset) / grid_size) if 0 <= x < 19 and 0 <= y < 19 and self.board[y][x] == 0: current_player_before_move = self.current_player # 保存当前玩家 self.place_piece(x, y) if self.check_win(x, y): self.game_over(current_player_before_move) # 传递正确玩家 elif self.mode == "ai": self.ai_move() def draw_board(self): offset = self.cell_size // 2 full_size = (self.board_size-1)*self.cell_size # 绘制网格 for i in range(self.board_size): y = i * self.cell_size + offset self.canvas.create_line(offset, y, full_size + offset, y) x = i * self.cell_size + offset self.canvas.create_line(x, offset, x, full_size + offset) # 星位标记 star_positions = [(3,3), (3,15), (15,3), (15,15), (9,9)] for (x, y) in star_positions: cx = x * self.cell_size + offset cy = y * self.cell_size + offset self.canvas.create_oval(cx-3, cy-3, cx+3, cy+3, fill='black') def init_game(self): self.board = [[0 for _ in range(19)] for _ in range(19)] self.current_player = 1 self.game_active = False self.history = [] self.canvas.delete("piece") self.canvas.delete("winline") def start_new_game(self): self.init_game() if self.mode == "human": dice = random.randint(1,6) self.current_player = 1 if dice <=3 else 2 messagebox.showinfo("骰子结果", f"骰子点数: {dice}\n玩家{self.current_player}先手") self.game_active = True if self.mode == "ai" and self.current_player == 2: self.ai_move() def check_win(self, x, y): player = self.board[y][x] directions = [(1,0), (0,1), (1,1), (1,-1)] for dx, dy in directions: count = 1 # 正向检测 step = 1 while True: nx = x + dx * step ny = y + dy * step if 0 <= nx < 19 and 0 <= ny < 19 and self.board[ny][nx] == player: count += 1 step += 1 else: break # 反向检测 step = 1 while True: nx = x - dx * step ny = y - dy * step if 0 <= nx < 19 and 0 <= ny < 19 and self.board[ny][nx] == player: count += 1 step += 1 else: break if count >= 5: self.highlight_winning_line(x, y, dx, dy) self.game_active = False return True return False def highlight_winning_line(self, x, y, dx, dy): for i in range(-4, 1): points = [] for j in range(5): nx = x + (dx * (i + j)) ny = y + (dy * (i + j)) if 0 <= nx < 19 and 0 <= ny < 19: points.append((nx, ny)) if len(points) == 5: for (px, py) in points: cx = (px * self.cell_size) + self.cell_size cy = (py * self.cell_size) + self.cell_size self.canvas.create_oval(cx-15, cy-15, cx+15, cy+15, outline="red", width=3, tags="winline") break def place_piece(self, x, y): color = "black" if self.current_player == 1 else "white" cx = (x * self.cell_size) + self.cell_size cy = (y * self.cell_size) + self.cell_size piece_id = self.canvas.create_oval(cx-12, cy-12, cx+12, cy+12, fill=color, tags="piece") self.board[y][x] = self.current_player self.history.append((x, y, piece_id)) self.current_player = 2 if self.current_player == 1 else 1 def game_over(self, winner): self.game_active = False # 更新积分（添加有效性检查） if winner in [1, 2]: self.player_scores[winner-1] += 1 self.round_count += 1 self.update_score() # 获取当前规则所需胜局数 required_wins = self.match_rules.get(self.current_rule, 2) # 检查是否赢得比赛 if max(self.player_scores) >= required_wins: final_winner = "玩家1" if self.player_scores[0] > self.player_scores[1] else "玩家2" if messagebox.askyesno("游戏结束", f"{final_winner}赢得比赛!\n开始新游戏？"): self.player_scores = [0, 0] self.round_count = 0 self.start_new_game() else: # 触发惩罚 loser = 2 if winner == 1 else 1 self.show_punishment(loser) self.start_new_game() def show_punishment(self, loser): choice = messagebox.askyesno( "真心话大冒险", f"玩家{loser}请选择:\n是 - 真心话\n否 - 大冒险" ) if choice: question = random.choice([ "你与多少人发生过性关系？", "描述你最尴尬的性经历？", "你有没有出轨过？", "你最疯狂的性幻想是什么？", "你什么时候失去童贞的？", "你有没有发过裸照给别人？" ]) messagebox.showinfo("真心话", question) else: task = random.choice([ "给前任打电话说'我想念你的身体'", "在公共场所舔地板", "给赢家发送下体照片", "亲吻左边的人10秒钟", "让赢家打屁股10下", "在社交媒体发布'我爱爸爸'" ]) messagebox.showinfo("大冒险", task) def update_score(self): self.score_label.config( text=f"分数 | 玩家1: {self.player_scores[0]} 玩家2: {self.player_scores[1]}" ) self.root.update_idletasks() # 强制刷新界面 def undo_move(self): if len(self.history) >= (2 if self.mode == "ai" else 1): for _ in range(2 if self.mode == "ai" else 1): x, y, piece = self.history.pop() self.canvas.delete(piece) self.board[y][x] = 0 self.current_player = 1 if self.current_player == 2 else 1 def toggle_mode(self): self.mode = "ai" if self.mode == "human" else "human" self.mode_btn.config(text=f"切换为{'双人' if self.mode=='ai' else 'AI'}模式") self.start_new_game() def change_rule(self, event): rule_map = {"三局两胜": 2, "五局三胜": 3, "十局六胜": 5} self.current_rule = self.rule_combobox.get() self.match_rules = {self.current_rule: rule_map[self.current_rule]} self.update_score() def update_difficulty(self, value): self.ai_difficulty = int(value) def ai_move(self): if not self.game_active or self.ai_thinking: return self.ai_thinking = True self.root.config(cursor="watch") self.root.update() try: if self.ai_difficulty <= 3: self.random_ai() elif self.ai_difficulty <= 7: self.smart_ai() else: self.expert_ai() if self.check_win(*self.last_ai_move): self.game_over(2) finally: self.ai_thinking = False self.root.config(cursor="") self.root.update() def random_ai(self): empty = self.get_empty_positions() if empty: x, y = random.choice(empty) self.place_piece(x, y) self.last_ai_move = (x, y) def smart_ai(self): best_score = -float('inf') best_moves = [] for y in range(19): for x in range(19): if self.board[y][x] == 0: attack = self.evaluate_position(x, y, 2) defend = self.evaluate_position(x, y, 1) total = attack * 0.7 + defend * 0.3 if total > best_score: best_score = total best_moves = [(x, y)] elif total == best_score: best_moves.append((x, y)) if best_moves: x, y = random.choice(best_moves) self.place_piece(x, y) self.last_ai_move = (x, y) def expert_ai(self): best_move = None best_score = -float('inf') start_time = time.time() search_depth = min(2 + self.ai_difficulty//3, 4) candidates = self.get_candidate_moves() for (x, y) in candidates: self.board[y][x] = 2 score = self.minimax(search_depth-1, False, -float('inf'), float('inf')) self.board[y][x] = 0 if score > best_score: best_score = score best_move = (x, y) if time.time() - start_time > 2: break if best_move: self.place_piece(*best_move) self.last_ai_move = best_move def minimax(self, depth, is_max, alpha, beta): if depth == 0: return self.evaluate_board() if is_max: max_score = -float('inf') for (x, y) in self.get_candidate_moves(): self.board[y][x] = 2 score = self.minimax(depth-1, False, alpha, beta) self.board[y][x] = 0 max_score = max(max_score, score) alpha = max(alpha, score) if beta <= alpha: break return max_score else: min_score = float('inf') for (x, y) in self.get_candidate_moves(): self.board[y][x] = 1 score = self.minimax(depth-1, True, alpha, beta) self.board[y][x] = 0 min_score = min(min_score, score) beta = min(beta, score) if beta <= alpha: break return min_score def evaluate_position(self, x, y, player): score = 0 directions = [(1,0), (0,1), (1,1), (1,-1)] for dx, dy in directions: line = [] for i in range(-4, 5): nx, ny = x + dx*i, y + dy*i if 0 <= nx < 19 and 0 <= ny < 19: line.append(self.board[ny][nx]) else: line.append(-1) for i in range(5): segment = line[i:i+5] if -1 in segment: continue count = sum(1 for p in segment if p == player) empty = sum(1 for p in segment if p == 0) if count == 4 and empty == 1: score += 5000 elif count == 3 and empty == 2: score += 1000 elif count == 2 and empty == 3: score += 300 return score def get_candidate_moves(self): candidates = set() for y in range(19): for x in range(19): if self.board[y][x] != 0: for dy in range(-2, 3): for dx in range(-2, 3): nx, ny = x+dx, y+dy if 0 <= nx < 19 and 0 <= ny < 19: if self.board[ny][nx] == 0: candidates.add((nx, ny)) return list(candidates) if candidates else self.get_empty_positions() def get_empty_positions(self): return [(x, y) for y in range(19) for x in range(19) if self.board[y][x] == 0] def run(self): self.root.mainloop() if __name__ == "__main__": game = Gobang() game.run()
