黑棋先手
// 游戏主逻辑 canvas.addEventListener('click', (e) => { if(gameState.gameOver) return; const pos = getBoardPosition(e.clientX, e.clientY); const x = pos.x, y = pos.y; if(x >= 0 && x < BOARD_SIZE && y >= 0 && y < BOARD_SIZE && gameState.board[y][x] === 0) { // 玩家落子 gameState.board[y][x] = gameState.currentPlayer; gameState.history.push({x, y, player: gameState.currentPlayer}); drawStone(x, y, gameState.currentPlayer); if(checkWin(x, y)) { gameState.gameOver = true; document.getElementById('message').textContent = `玩家${gameState.currentPlayer}获胜！`; return; } if(gameState.aiEnabled) { // AI回合 setTimeout(() => { const ai = new AI(gameState.aiDifficulty); const [aiX, aiY] = ai.getMove(); if(gameState.board[aiY][aiX] === 0) { gameState.board[aiY][aiX] = 2; gameState.history.push({x: aiX, y: aiY, player: 2}); drawStone(aiX, aiY, 2); if(checkWin(aiX, aiY)) { gameState.gameOver = true; document.getElementById('message').textContent = "AI获胜！"; } else { gameState.currentPlayer = 1; document.getElementById('message').textContent = "黑棋回合"; } } }, 500); } else { // 双人模式 gameState.currentPlayer = 3 - gameState.currentPlayer; document.getElementById('message').textContent = `${gameState.currentPlayer === 1 ? '黑棋' : '白棋'}回合`; } } }); // 初始化游戏 initBoard(); // 事件监听 document.getElementById('gameMode').addEventListener('change', (e) => { gameState.aiEnabled = e.target.value === 'ai'; gameState = { board: Array(BOARD_SIZE).fill().map(() => Array(BOARD_SIZE).fill(0)), currentPlayer: 1, gameOver: false, history: [], aiEnabled: gameState.aiEnabled, aiDifficulty: gameState.aiDifficulty }; initBoard(); document.getElementById('message').textContent = "黑棋先手"; }); document.getElementById('aiDifficulty').addEventListener('input', (e) => { gameState.aiDifficulty = parseInt(e.target.value); document.getElementById('difficultyValue').textContent = e.target.value; }); document.getElementById('resetButton').addEventListener('click', () => { gameState = { board: Array(BOARD_SIZE).fill().map(() => Array(BOARD_SIZE).fill(0)), currentPlayer: 1, gameOver: false, history: [], aiEnabled: gameState.aiEnabled, aiDifficulty: gameState.aiDifficulty }; initBoard(); document.getElementById('message').textContent = "黑棋先手"; }); document.getElementById('undoBtn').addEventListener('click', () => { if(gameState.history.length > 0) { const lastMove = gameState.history.pop(); gameState.board[lastMove.y][lastMove.x] = 0; gameState.currentPlayer = lastMove.player; initBoard(); document.getElementById('message').textContent = `${gameState.currentPlayer === 1 ? '黑棋' : '白棋'}回合`; } }); // 键盘事件 document.addEventListener('keydown', (e) => { if(e.ctrlKey && e.key === 'z') { document.getElementById('undoBtn').click(); } });
